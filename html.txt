            var date1Mago = new Date(strToday.substring(0, 4),strToday.substring(4, 6) - 7,strToday.substring(6, 8));

        function gotoPrewPage() {

            var strReqKey2 = "";
            var strBackKey1 = "";

            var pageNum = parseInt('0');

                IBF.elements['다음거래일련번호키'].value = "";
            } else {
                var nLastKeyPoint1 = strBackKey1.lastIndexOf("|");
                var nLastKeyPoint2 = strBackKey2.lastIndexOf("|");
                strReqKey1 = strBackKey1.substring(nLastKeyPoint1 + 1, strBackKey1.length);
                strReqKey2 = strBackKey2.substring(nLastKeyPoint2 + 1, strBackKey2.length);
                strBackKey1 = strBackKey1.substring(0, nLastKeyPoint1);
                strBackKey2 = strBackKey2.substring(0, nLastKeyPoint2);
                IBF.elements['다음거래년월일키'].value = strReqKey1;

                IBF.elements['이전다음거래일련번호키'].value = strBackKey2;
